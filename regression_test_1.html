  let currentIndex = 0;


  function renderCurrentTest() {

    const container = document.getElementById("current-test");

    const t = tests[currentIndex];


    const total = tests.length;

    const status = t.status || "pending";

    const comment = t.comment || "";


    let statusLabel = "Pending";

    let statusClass = "status-pending";

    if (status === "pass") {

      statusLabel = "Passed";

      statusClass = "status-pass";

    } else if (status === "fail") {

      statusLabel = "Failed";

      statusClass = "status-fail";

    }


    container.innerHTML = `

      <div class="test-header">

        <div>

          <div class="section-label">${t.section}</div>

          <div class="test-title">${t.id} – ${t.title}</div>

        </div>

        <div style="text-align:right;">

          <div class="progress">Test ${currentIndex + 1} of ${total}</div>

          <div class="status-pill ${statusClass}">${statusLabel}</div>

        </div>

      </div>

      <div class="test-details">${escapeHtml(t.details)}</div>

      <div class="comment-label">

        Comment (required if this test fails):

      </div>

      <textarea id="comment-input" placeholder="Add notes here for failures, edge cases, or anything unexpected...">${escapeHtml(comment)}</textarea>

      <div id="error-message" class="error"></div>

      <div class="actions">

        <button type="button" class="btn-pass" onclick="markAndNext('pass')">

          ✅ Mark Pass & Next

        </button>

        <button type="button" class="btn-fail" onclick="markAndNext('fail')">

          ❌ Mark Fail & Next

        </button>

        <button type="button" class="btn-secondary" onclick="goPrev()" ${currentIndex === 0 ? "disabled" : ""}>

          ← Previous

        </button>

        <button type="button" class="btn-secondary" onclick="goNext()" ${currentIndex === total - 1 ? "disabled" : ""}>

          Skip / Next →

        </button>

      </div>

    `;

  }


  function markAndNext(status) {

    const textarea = document.getElementById("comment-input");

    const errorEl = document.getElementById("error-message");

    const comment = textarea.value.trim();


    if (status === "fail" && comment.length === 0) {

      errorEl.textContent = "Please add a brief comment before marking this test as failed.";

      return;

    }

    errorEl.textContent = "";


    tests[currentIndex].status = status;

    tests[currentIndex].comment = comment;


    if (currentIndex < tests.length - 1) {

      currentIndex++;

      renderCurrentTest();

    } else {

      // Last test: just re-render to update status

      renderCurrentTest();

      alert("You’ve reached the last test. Check the Summary section below for an overview.");

    }


    renderSummary();

  }


  function goPrev() {

    if (currentIndex > 0) {

      // Save comment but do not change status

      const textarea = document.getElementById("comment-input");

      if (textarea) {

        tests[currentIndex].comment = textarea.value;

      }

      currentIndex--;

      renderCurrentTest();

    }

  }


  function goNext() {

    if (currentIndex < tests.length - 1) {

      // Save comment but do not change status

      const textarea = document.getElementById("comment-input");

      if (textarea) {

        tests[currentIndex].comment = textarea.value;

      }

      currentIndex++;

      renderCurrentTest();

    }

  }


  function renderSummary() {

    const wrapper = document.getElementById("summary-table-wrapper");

    const rows = tests.map(t => {

      const status = t.status || "pending";

      let label = "Pending";

      let statusClass = "status-pending";

      if (status === "pass") {

        label = "Pass";

        statusClass = "status-pass";

      } else if (status === "fail") {

        label = "Fail";

        statusClass = "status-fail";

      }

      return `

        <tr>

          <td class="summary-section">${t.section}<br><strong>${t.id}</strong></td>

          <td>${escapeHtml(t.title)}</td>

          <td class="summary-status">

            <span class="status-pill ${statusClass}">${label}</span>

          </td>

          <td class="summary-comment">${t.comment ? escapeHtml(t.comment) : ""}</td>

        </tr>

      `;

    }).join("");


    wrapper.innerHTML = `

      <table>

        <thead>

          <tr>

            <th>Section / ID</th>

            <th>Test</th>

            <th>Status</th>

            <th>Comments (on Fail)</th>

          </tr>

        </thead>

        <tbody>

          ${rows}

        </tbody>

      </table>

    `;

  }


  function escapeHtml(str) {

    if (!str) return "";

    return String(str)

      .replace(/&/g, "&amp;")

      .replace(/</g, "&lt;")

      .replace(/>/g, "&gt;")

      .replace(/"/g, "&quot;");

  }


  // Initial render

  renderCurrentTest();

  renderSummary();
